    iter = [
        pl.col("timestamp").min().alias("created_on"),

        colts.max().alias('max'),
        colts.count().alias("count"),
        pl.col("event").filter(pl.col("event") == "Modify").count().alias("modify"),
    ]
    df = pl.DataFrame()
    for i in iter:
        _df=df2.lazy().group_by(pl.col("order_id")).agg([i])

        schema = [func("order_id"), func("order_book_id"), func("side")]

def cell():
    func = lambda s: pl.col(s)
    func2 = lambda s: pl.col("order_id").struct.field(s).alias(s)
    func3 = lambda s: pl.col("order_id").struct.field(s)
    schema = [func("order_id"), func("order_book_id"), func("side")]
    unnest = [func2("order_id"), func2("order_book_id"), func2("side")]
    schema3 = [func3("order_id"), func3("order_book_id"), func3("side")]

    m_df2_2=df2.filter(pl.col("event") == "Modify").group_by(pl.col("order_id")).count().with_columns(pl.col("count").alias("modify_count"))
    d_df2_2=df2.filter(pl.col("event") == "Deleted").group_by(pl.col("order_id")).count().filter(pl.col("count") > 0).with_columns(pl.col("count").cast(pl.Boolean).alias("is_deleted"))
    df2_2=m_df2_2.join(d_df2_2, on=schema3, how="inner").select(pl.exclude(["count_right", "count"]))
    # print(df2_2)
    # print(df2)
    _df= df2.with_columns(unnest).lazy().group_by(schema).agg([pl.col("timestamp").sort()]).join(df2_2.lazy(), on=schema)
    _df=_df.with_columns(pl.col("timestamp").list.sort().list.diff().list.min().alias("min_time"))
    _df=_df.with_columns([pl.col("timestamp").list.first().alias("created_on"), pl.col("timestamp").list.last().alias("last_observation"), pl.col("timestamp").list.len().alias("event_count")])
    _df=_df.collect(streaming=True).select(pl.exclude(pl.col("timestamp")))
    return _df.sort("min_time")
